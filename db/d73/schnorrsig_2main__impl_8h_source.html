107 static int secp256k1_schnorrsig_real_verify(const secp256k1_context* ctx, secp256k1_gej *rj, const secp256k1_scalar *s, const secp256k1_scalar *e, const secp256k1_pubkey *pk) {

126  secp256k1_scalar e;
127  secp256k1_gej rj;
128  secp256k1_fe rx;
129  secp256k1_sha256 sha;
130  unsigned char buf[33];
131  size_t buflen = sizeof(buf);
132  int overflow;
133 
134  VERIFY_CHECK(ctx != NULL);
135  ARG_CHECK(secp256k1_ecmult_context_is_built(&ctx->ecmult_ctx));
136  ARG_CHECK(sig != NULL);
137  ARG_CHECK(msg32 != NULL);
138  ARG_CHECK(pk != NULL);

140  if (!secp256k1_fe_set_b32(&rx, &sig->data[0])) {
141  return 0;
142  }
143 
144  secp256k1_scalar_set_b32(&s, &sig->data[32], &overflow);
145  if (overflow) {
146  return 0;
147  }
148 
149  secp256k1_sha256_initialize(&sha);
150  secp256k1_sha256_write(&sha, &sig->data[0], 32);
151  secp256k1_ec_pubkey_serialize(ctx, buf, &buflen, pk, SECP256K1_EC_COMPRESSED);
152  secp256k1_sha256_write(&sha, buf, buflen);
153  secp256k1_sha256_write(&sha, msg32, 32);
154  secp256k1_sha256_finalize(&sha, buf);
155  secp256k1_scalar_set_b32(&e, buf, NULL);
156 
157  if (!secp256k1_schnorrsig_real_verify(ctx, &rj, &s, &e, pk)
158  || !secp256k1_gej_has_quad_y_var(&rj) /* fails if rj is infinity */
159  || !secp256k1_gej_eq_x_var(&rx, &rj)) {
160  return 0;
161  }
162 
163  return 1;
164 }

167 typedef struct {
168  const secp256k1_context *ctx;
169  /* Seed for the random number generator */
170  unsigned char chacha_seed[32];
171  /* Caches randomizers generated by the PRNG which returns two randomizers per call. Caching

173  * the PRNG is called at every odd indexed schnorrsig to fill the cache. */
174  secp256k1_scalar randomizer_cache[2];
175  /* Signature, message, public key tuples to verify */
176  const secp256k1_schnorrsig *const *sig;
177  const unsigned char *const *msg32;
178  const secp256k1_pubkey *const *pk;
179  size_t n_sigs;
180 } secp256k1_schnorrsig_verify_ecmult_context;

182 /* Callback function which is called by ecmult_multi in order to convert the ecmult_context
183  * consisting of signature, message and public key tuples into scalars and points. */
184 static int secp256k1_schnorrsig_verify_batch_ecmult_callback(secp256k1_scalar *sc, secp256k1_ge *pt, size_t idx, void *data) {
185  secp256k1_schnorrsig_verify_ecmult_context *ecmult_context = (secp256k1_schnorrsig_verify_ecmult_context *) data;
186 
187  if (idx % 4 == 2) {
188  /* Every idx corresponds to a (scalar,point)-tuple. So this callback is called with 4
189  * consecutive tuples before we need to call the RNG for new randomizers:
190  * (-randomizer_cache[0], R1)
191  * (-randomizer_cache[0]*e1, P1)
192  * (-randomizer_cache[1], R2)
193  * (-randomizer_cache[1]*e2, P2) */
194  secp256k1_scalar_chacha20(&ecmult_context->randomizer_cache[0], &ecmult_context->randomizer_cache[1], ecmult_context->chacha_seed, idx / 4);
195  }
196 
197  /* R */
198  if (idx % 2 == 0) {
199  secp256k1_fe rx;
200  *sc = ecmult_context->randomizer_cache[(idx / 2) % 2];
201  if (!secp256k1_fe_set_b32(&rx, &ecmult_context->sig[idx / 2]->data[0])) {
202  return 0;
203  }
204  if (!secp256k1_ge_set_xquad(pt, &rx)) {
205  return 0;
206  }
207  /* eP */
208  } else {
209  unsigned char buf[33];
210  size_t buflen = sizeof(buf);
211  secp256k1_sha256 sha;
212  secp256k1_sha256_initialize(&sha);
213  secp256k1_sha256_write(&sha, &ecmult_context->sig[idx / 2]->data[0], 32);
214  secp256k1_ec_pubkey_serialize(ecmult_context->ctx, buf, &buflen, ecmult_context->pk[idx / 2], SECP256K1_EC_COMPRESSED);
215  secp256k1_sha256_write(&sha, buf, buflen);
216  secp256k1_sha256_write(&sha, ecmult_context->msg32[idx / 2], 32);
217  secp256k1_sha256_finalize(&sha, buf);
218 
219  secp256k1_scalar_set_b32(sc, buf, NULL);
220  secp256k1_scalar_mul(sc, sc, &ecmult_context->randomizer_cache[(idx / 2) % 2]);
221 
222  if (!secp256k1_pubkey_load(ecmult_context->ctx, pt, ecmult_context->pk[idx / 2])) {
223  return 0;
224  }
225  }
226  return 1;
227 }

279 static int secp256k1_schnorrsig_verify_batch_sum_s(secp256k1_scalar *s, unsigned char *chacha_seed, const secp256k1_schnorrsig *const *sig, size_t n_sigs) {
280  secp256k1_scalar randomizer_cache[2];
281  size_t i;
282 
283  secp256k1_scalar_set_int(&randomizer_cache[0], 1);
284  for (i = 0; i < n_sigs; i++) {
285  int overflow;
286  secp256k1_scalar term;
287  if (i % 2 == 1) {
288  secp256k1_scalar_chacha20(&randomizer_cache[0], &randomizer_cache[1], chacha_seed, i / 2);
289  }
290 
291  secp256k1_scalar_set_b32(&term, &sig[i]->data[32], &overflow);
292  if (overflow) {
293  return 0;
294  }
295  secp256k1_scalar_mul(&term, &term, &randomizer_cache[i % 2]);
296  secp256k1_scalar_add(s, s, &term);
297  }
298  return 1;
299 }

301 /* schnorrsig batch verification.
302  * Seeds a random number generator with the inputs and derives a random number ai for every
303  * signature i. Fails if y-coordinate of any R is not a quadratic residue or if
304  * 0 != -(s1 + a2*s2 + ... + au*su)G + R1 + a2*R2 + ... + au*Ru + e1*P1 + (a2*e2)P2 + ... + (au*eu)Pu. */
305 int secp256k1_schnorrsig_verify_batch(const secp256k1_context *ctx, secp256k1_scratch *scratch, const secp256k1_schnorrsig *const *sig, const unsigned char *const *msg32, const secp256k1_pubkey *const *pk, size_t n_sigs) {
306  secp256k1_schnorrsig_verify_ecmult_context ecmult_context;
307  secp256k1_sha256 sha;
308  secp256k1_scalar s;
309  secp256k1_gej rj;
310 
311  VERIFY_CHECK(ctx != NULL);
312  ARG_CHECK(secp256k1_ecmult_context_is_built(&ctx->ecmult_ctx));
313  ARG_CHECK(scratch != NULL);
314  /* Check that n_sigs is less than half of the maximum size_t value. This is necessary because
315  * the number of points given to ecmult_multi is 2*n_sigs. */
316  ARG_CHECK(n_sigs <= SIZE_MAX / 2);
317  /* Check that n_sigs is less than 2^31 to ensure the same behavior of this function on 32-bit
318  * and 64-bit platforms. */
319  ARG_CHECK(n_sigs < ((uint32_t)1 << 31));
320 
321  secp256k1_sha256_initialize(&sha);
322  if (!secp256k1_schnorrsig_verify_batch_init_randomizer(ctx, &ecmult_context, &sha, sig, msg32, pk, n_sigs)) {
323  return 0;
324  }
325  secp256k1_sha256_finalize(&sha, ecmult_context.chacha_seed);
326  secp256k1_scalar_set_int(&ecmult_context.randomizer_cache[0], 1);
327 
328  secp256k1_scalar_clear(&s);
329  if (!secp256k1_schnorrsig_verify_batch_sum_s(&s, ecmult_context.chacha_seed, sig, n_sigs)) {
330  return 0;
331  }
332  secp256k1_scalar_negate(&s, &s);
333 
334  return secp256k1_ecmult_multi_var(&ctx->ecmult_ctx, scratch, &rj, &s, secp256k1_schnorrsig_verify_batch_ecmult_callback, (void *) &ecmult_context, 2 * n_sigs)
335  && secp256k1_gej_is_infinity(&rj);
336 }
