227  int nVersion = s.GetVersion();
228  if (!(s.GetType() & SER_GETHASH))
229  READWRITE(nVersion);

234  READWRITE(fInternal);
235  }
236  catch (std::ios_base::failure&) {
237  /* flag as external address if we can't read the internal boolean
238  (this will be the case for any wallet before the HD chain split version) */
239  fInternal = false;
240  }

242  READWRITE(m_pre_split);
243  }
244  catch (std::ios_base::failure&) {
245  /* flag as postsplit address if we can't read the m_pre_split boolean
246  (this will be the case for any wallet that upgrades to HD chain split)*/
247  m_pre_split = false;
248  }
249  }

251  READWRITE(fInternal);
