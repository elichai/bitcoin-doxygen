187 // Require that user allocate at least 550 MiB for block & undo files (blk???.dat and rev???.dat)
188 // At 1MB per block, 288 blocks = 288MB.
189 // Add 15% for Undo data = 331MB
190 // Add 20% for Orphan block rate = 397MB
191 // We want the low water mark after pruning to be at least 397 MB and since we prune in
192 // full block file chunks, we need the high water mark which triggers the prune to be
193 // one 128MB block file + added 15% undo data = 147MB greater for a total of 545MB
194 // Setting the target to >= 550 MiB will make it likely we can respect the target.
195 static const uint64_t MIN_DISK_SPACE_FOR_BLOCK_FILES = 550 * 1024 * 1024;
