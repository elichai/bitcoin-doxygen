64  template <typename Stream>
65  inline void Serialize(Stream& s) const {
66  // Write the utxo
67  // If there is a non-witness utxo, then don't add the witness one.
68  if (non_witness_utxo) {
69  SerializeToVector(s, PSBT_IN_NON_WITNESS_UTXO);
70  OverrideStream<Stream> os(&s, s.GetType(), s.GetVersion() | SERIALIZE_TRANSACTION_NO_WITNESS);
71  SerializeToVector(os, non_witness_utxo);
72  } else if (!witness_utxo.IsNull()) {
73  SerializeToVector(s, PSBT_IN_WITNESS_UTXO);
74  SerializeToVector(s, witness_utxo);
75  }
76 
77  if (final_script_sig.empty() && final_script_witness.IsNull()) {
78  // Write any partial signatures
79  for (auto sig_pair : partial_sigs) {
80  SerializeToVector(s, PSBT_IN_PARTIAL_SIG, MakeSpan(sig_pair.second.first));
81  s << sig_pair.second.second;
82  }
83 
84  // Write the sighash type
85  if (sighash_type > 0) {
86  SerializeToVector(s, PSBT_IN_SIGHASH);
87  SerializeToVector(s, sighash_type);
88  }
89 
90  // Write the redeem script
91  if (!redeem_script.empty()) {
92  SerializeToVector(s, PSBT_IN_REDEEMSCRIPT);
93  s << redeem_script;
94  }
95 
96  // Write the witness script
97  if (!witness_script.empty()) {
98  SerializeToVector(s, PSBT_IN_WITNESSSCRIPT);
99  s << witness_script;
100  }
101 
102  // Write any hd keypaths
103  SerializeHDKeypaths(s, hd_keypaths, PSBT_IN_BIP32_DERIVATION);
104  }
105 
106  // Write script sig
107  if (!final_script_sig.empty()) {
108  SerializeToVector(s, PSBT_IN_SCRIPTSIG);
109  s << final_script_sig;
110  }
111  // write script witness
112  if (!final_script_witness.IsNull()) {
113  SerializeToVector(s, PSBT_IN_SCRIPTWITNESS);
114  SerializeToVector(s, final_script_witness.stack);
115  }
116 
117  // Write unknown things
118  for (auto& entry : unknown) {
119  s << entry.first;
120  s << entry.second;
121  }
122 
123  s << PSBT_SEPARATOR;
124  }

127  template <typename Stream>
128  inline void Unserialize(Stream& s) {
129  // Read loop
130  bool found_sep = false;
131  while(!s.empty()) {
132  // Read
133  std::vector<unsigned char> key;
134  s >> key;
135 
136  // the key is empty if that was actually a separator byte
137  // This is a special case for key lengths 0 as those are not allowed (except for separator)
138  if (key.empty()) {
139  found_sep = true;
140  break;
141  }
142 
143  // First byte of key is the type
144  unsigned char type = key[0];
145 
146  // Do stuff based on type
147  switch(type) {
148  case PSBT_IN_NON_WITNESS_UTXO:
149  {
150  if (non_witness_utxo) {
151  throw std::ios_base::failure("Duplicate Key, input non-witness utxo already provided");
152  } else if (key.size() != 1) {
153  throw std::ios_base::failure("Non-witness utxo key is more than one byte type");
154  }
155  // Set the stream to unserialize with witness since this is always a valid network transaction
156  OverrideStream<Stream> os(&s, s.GetType(), s.GetVersion() & ~SERIALIZE_TRANSACTION_NO_WITNESS);
157  UnserializeFromVector(os, non_witness_utxo);
158  break;
159  }
160  case PSBT_IN_WITNESS_UTXO:
161  if (!witness_utxo.IsNull()) {
162  throw std::ios_base::failure("Duplicate Key, input witness utxo already provided");
163  } else if (key.size() != 1) {
164  throw std::ios_base::failure("Witness utxo key is more than one byte type");
165  }
166  UnserializeFromVector(s, witness_utxo);
167  break;
168  case PSBT_IN_PARTIAL_SIG:
169  {
170  // Make sure that the key is the size of pubkey + 1
171  if (key.size() != CPubKey::PUBLIC_KEY_SIZE + 1 && key.size() != CPubKey::COMPRESSED_PUBLIC_KEY_SIZE + 1) {
172  throw std::ios_base::failure("Size of key was not the expected size for the type partial signature pubkey");
173  }
174  // Read in the pubkey from key
175  CPubKey pubkey(key.begin() + 1, key.end());
176  if (!pubkey.IsFullyValid()) {
177  throw std::ios_base::failure("Invalid pubkey");
178  }
179  if (partial_sigs.count(pubkey.GetID()) > 0) {
180  throw std::ios_base::failure("Duplicate Key, input partial signature for pubkey already provided");
181  }
182 
183  // Read in the signature from value
184  std::vector<unsigned char> sig;
185  s >> sig;
186 
187  // Add to list
188  partial_sigs.emplace(pubkey.GetID(), SigPair(pubkey, std::move(sig)));
189  break;
190  }
191  case PSBT_IN_SIGHASH:
192  if (sighash_type > 0) {
193  throw std::ios_base::failure("Duplicate Key, input sighash type already provided");
194  } else if (key.size() != 1) {
195  throw std::ios_base::failure("Sighash type key is more than one byte type");
196  }
197  UnserializeFromVector(s, sighash_type);
198  break;
199  case PSBT_IN_REDEEMSCRIPT:
200  {
201  if (!redeem_script.empty()) {
202  throw std::ios_base::failure("Duplicate Key, input redeemScript already provided");
203  } else if (key.size() != 1) {
204  throw std::ios_base::failure("Input redeemScript key is more than one byte type");
205  }
206  s >> redeem_script;
207  break;
208  }
209  case PSBT_IN_WITNESSSCRIPT:
210  {
211  if (!witness_script.empty()) {
212  throw std::ios_base::failure("Duplicate Key, input witnessScript already provided");
213  } else if (key.size() != 1) {
214  throw std::ios_base::failure("Input witnessScript key is more than one byte type");
215  }
216  s >> witness_script;
217  break;
218  }
219  case PSBT_IN_BIP32_DERIVATION:
220  {
221  DeserializeHDKeypaths(s, key, hd_keypaths);
222  break;
223  }
224  case PSBT_IN_SCRIPTSIG:
225  {
226  if (!final_script_sig.empty()) {
227  throw std::ios_base::failure("Duplicate Key, input final scriptSig already provided");
228  } else if (key.size() != 1) {
229  throw std::ios_base::failure("Final scriptSig key is more than one byte type");
230  }
231  s >> final_script_sig;
232  break;
233  }
234  case PSBT_IN_SCRIPTWITNESS:
235  {
236  if (!final_script_witness.IsNull()) {
237  throw std::ios_base::failure("Duplicate Key, input final scriptWitness already provided");
238  } else if (key.size() != 1) {
239  throw std::ios_base::failure("Final scriptWitness key is more than one byte type");
240  }
241  UnserializeFromVector(s, final_script_witness.stack);
242  break;
243  }
244  // Unknown stuff
245  default:
246  if (unknown.count(key) > 0) {
247  throw std::ios_base::failure("Duplicate Key, key for unknown value already provided");
248  }
249  // Read in the value
250  std::vector<unsigned char> val_bytes;
251  s >> val_bytes;
252  unknown.emplace(std::move(key), std::move(val_bytes));
253  break;
254  }
255  }
256 
257  if (!found_sep) {
258  throw std::ios_base::failure("Separator is missing at the end of an input map");
259  }
260  }
261 
262  template <typename Stream>
263  PSBTInput(deserialize_type, Stream& s) {
264  Unserialize(s);
265  }

442  template <typename Stream>
443  inline void Unserialize(Stream& s) {
444  // Read the magic bytes
445  uint8_t magic[5];
446  s >> magic;
447  if (!std::equal(magic, magic + 5, PSBT_MAGIC_BYTES)) {
448  throw std::ios_base::failure("Invalid PSBT magic bytes");
449  }
450 
451  // Read global data
452  bool found_sep = false;
453  while(!s.empty()) {
454  // Read
455  std::vector<unsigned char> key;
456  s >> key;
457 
458  // the key is empty if that was actually a separator byte
459  // This is a special case for key lengths 0 as those are not allowed (except for separator)
460  if (key.empty()) {
461  found_sep = true;
462  break;
463  }
464 
465  // First byte of key is the type
466  unsigned char type = key[0];
467 
468  // Do stuff based on type
469  switch(type) {
470  case PSBT_GLOBAL_UNSIGNED_TX:
471  {
472  if (tx) {
473  throw std::ios_base::failure("Duplicate Key, unsigned tx already provided");
474  } else if (key.size() != 1) {
475  throw std::ios_base::failure("Global unsigned tx key is more than one byte type");
476  }
477  CMutableTransaction mtx;
478  // Set the stream to serialize with non-witness since this should always be non-witness
479  OverrideStream<Stream> os(&s, s.GetType(), s.GetVersion() | SERIALIZE_TRANSACTION_NO_WITNESS);
480  UnserializeFromVector(os, mtx);
481  tx = std::move(mtx);
482  // Make sure that all scriptSigs and scriptWitnesses are empty
483  for (const CTxIn& txin : tx->vin) {
484  if (!txin.scriptSig.empty() || !txin.scriptWitness.IsNull()) {
485  throw std::ios_base::failure("Unsigned tx does not have empty scriptSigs and scriptWitnesses.");
486  }
487  }
488  break;
489  }
490  // Unknown stuff
491  default: {
492  if (unknown.count(key) > 0) {
493  throw std::ios_base::failure("Duplicate Key, key for unknown value already provided");
494  }
495  // Read in the value
496  std::vector<unsigned char> val_bytes;
497  s >> val_bytes;
498  unknown.emplace(std::move(key), std::move(val_bytes));
499  }
500  }
501  }
502 
503  if (!found_sep) {
504  throw std::ios_base::failure("Separator is missing at the end of the global map");
505  }
506 
507  // Make sure that we got an unsigned tx
508  if (!tx) {
509  throw std::ios_base::failure("No unsigned transcation was provided");
510  }
511 
512  // Read input data
513  unsigned int i = 0;
514  while (!s.empty() && i < tx->vin.size()) {
515  PSBTInput input;
516  s >> input;
517  inputs.push_back(input);
518 
519  // Make sure the non-witness utxo matches the outpoint
520  if (input.non_witness_utxo && input.non_witness_utxo->GetHash() != tx->vin[i].prevout.hash) {
521  throw std::ios_base::failure("Non-witness UTXO does not match outpoint hash");
522  }
523  ++i;
524  }
525  // Make sure that the number of inputs matches the number of inputs in the transaction
526  if (inputs.size() != tx->vin.size()) {
527  throw std::ios_base::failure("Inputs provided does not match the number of inputs in transaction.");
528  }
529 
530  // Read output data
531  i = 0;
532  while (!s.empty() && i < tx->vout.size()) {
533  PSBTOutput output;
534  s >> output;
535  outputs.push_back(output);
536  ++i;
537  }
538  // Make sure that the number of outputs matches the number of outputs in the transaction
539  if (outputs.size() != tx->vout.size()) {
540  throw std::ios_base::failure("Outputs provided does not match the number of outputs in transaction.");
541  }
542  // Sanity check
543  if (!IsSane()) {
544  throw std::ios_base::failure("PSBT is not sane.");
545  }
546  }
