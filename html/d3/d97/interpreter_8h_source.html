44  // Evaluate P2SH subscripts (BIP16).
45  SCRIPT_VERIFY_P2SH = (1U << 0),
46 
47  // Passing a non-strict-DER signature or one with undefined hashtype to a checksig operation causes script failure.
48  // Evaluating a pubkey that is not (0x04 + 64 bytes) or (0x02 or 0x03 + 32 bytes) by checksig causes script failure.
49  // (not used or intended as a consensus rule).
50  SCRIPT_VERIFY_STRICTENC = (1U << 1),
51 
52  // Passing a non-strict-DER signature to a checksig operation causes script failure (BIP62 rule 1)
53  SCRIPT_VERIFY_DERSIG = (1U << 2),
54 
55  // Passing a non-strict-DER signature or one with S > order/2 to a checksig operation causes script failure
56  // (BIP62 rule 5).
57  SCRIPT_VERIFY_LOW_S = (1U << 3),
58 
59  // verify dummy stack item consumed by CHECKMULTISIG is of zero-length (BIP62 rule 7).
60  SCRIPT_VERIFY_NULLDUMMY = (1U << 4),
61 
62  // Using a non-push operator in the scriptSig causes script failure (BIP62 rule 2).
63  SCRIPT_VERIFY_SIGPUSHONLY = (1U << 5),
64 
65  // Require minimal encodings for all push operations (OP_0... OP_16, OP_1NEGATE where possible, direct
66  // pushes up to 75 bytes, OP_PUSHDATA up to 255 bytes, OP_PUSHDATA2 for anything larger). Evaluating
67  // any other push causes the script to fail (BIP62 rule 3).
68  // In addition, whenever a stack element is interpreted as a number, it must be of minimal length (BIP62 rule 4).
69  SCRIPT_VERIFY_MINIMALDATA = (1U << 6),
70 
71  // Discourage use of NOPs reserved for upgrades (NOP1-10)
72  //
73  // Provided so that nodes can avoid accepting or mining transactions
74  // containing executed NOP's whose meaning may change after a soft-fork,
75  // thus rendering the script invalid; with this flag set executing
76  // discouraged NOPs fails the script. This verification flag will never be
77  // a mandatory flag applied to scripts in a block. NOPs that are not
78  // executed, e.g. within an unexecuted IF ENDIF block, are *not* rejected.
79  // NOPs that have associated forks to give them new meaning (CLTV, CSV)
80  // are not subject to this rule.
81  SCRIPT_VERIFY_DISCOURAGE_UPGRADABLE_NOPS = (1U << 7),
82 
83  // Require that only a single stack element remains after evaluation. This changes the success criterion from
84  // "At least one stack element must remain, and when interpreted as a boolean, it must be true" to
85  // "Exactly one stack element must remain, and when interpreted as a boolean, it must be true".
86  // (BIP62 rule 6)
87  // Note: CLEANSTACK should never be used without P2SH or WITNESS.
88  SCRIPT_VERIFY_CLEANSTACK = (1U << 8),
89 
90  // Verify CHECKLOCKTIMEVERIFY
91  //
92  // See BIP65 for details.
93  SCRIPT_VERIFY_CHECKLOCKTIMEVERIFY = (1U << 9),
94 
95  // support CHECKSEQUENCEVERIFY opcode
96  //
97  // See BIP112 for details
98  SCRIPT_VERIFY_CHECKSEQUENCEVERIFY = (1U << 10),
99 
100  // Support segregated witness
101  //
102  SCRIPT_VERIFY_WITNESS = (1U << 11),
103 
104  // Making v1-v16 witness program non-standard
105  //
106  SCRIPT_VERIFY_DISCOURAGE_UPGRADABLE_WITNESS_PROGRAM = (1U << 12),
107 
108  // Segwit script only: Require the argument of OP_IF/NOTIF to be exactly 0x01 or empty vector
109  //
110  SCRIPT_VERIFY_MINIMALIF = (1U << 13),
111 
112  // Signature(s) must be empty vector if a CHECK(MULTI)SIG operation failed
113  //
114  SCRIPT_VERIFY_NULLFAIL = (1U << 14),
115 
116  // Public keys in segregated witness scripts must be compressed
117  //
118  SCRIPT_VERIFY_WITNESS_PUBKEYTYPE = (1U << 15),
119
