150  uint64_t randbits(int bits) noexcept {
151  if (bits == 0) {
152  return 0;
153  } else if (bits > 32) {
154  return rand64() >> (64 - bits);
155  } else {
156  if (bitbuf_size < bits) FillBitBuffer();
157  uint64_t ret = bitbuf & (~(uint64_t)0 >> (64 - bits));
158  bitbuf >>= bits;
159  bitbuf_size -= bits;
160  return ret;
161  }
